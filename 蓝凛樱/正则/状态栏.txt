    function SafeGetValue(obj, path, defaultValue = "N/A") {
        let keys = Array.isArray(path) ? path : path.split('.');
        let current = obj;
        for (let i = 0; i < keys.length; i++) {
            const key = keys[i];
            if (current === undefined || current === null || typeof current !== 'object' || !current.hasOwnProperty(key)) {
                return defaultValue;
            }
            current = current[key];
        }
        if (current === undefined || current === null) {
            return defaultValue;
        }
        // If SafeGetValue is expected to return the array itself for updateProgressBar
        // then remove the Array.isArray check from here, or pass the whole 'current'
        // Let updateProgressBar handle if rawValueFromCharacterData is an array
        return current; // Return the value directly, could be array, string, number etc.
    }

    function updateProgressBar(barId, valueMainId, valueDescId, rawValueFromCharacterData, min, max, color) {
        const valueMainDisplay = document.getElementById(valueMainId);
        const valueDescDisplay = document.getElementById(valueDescId);

        if (!valueMainDisplay) {
            return;
        }

        let workString;

        if (Array.isArray(rawValueFromCharacterData)) {
            workString = String(rawValueFromCharacterData[0]);
            // The description from rawValueFromCharacterData[1] is NOT displayed here
            // by this function in valueDescDisplay. That span is cleared.
        } else {
            workString = String(rawValueFromCharacterData);
        }

        // Remove trailing " (description)" from workString itself
        const descMatch = workString.match(/\s*\((.*)\)$/);
        if (descMatch) {
            workString = workString.substring(0, descMatch.index).trim();
        }

        // Handle "->" to get the final display value
        const parts = workString.split('->');
        let finalDisplayValue;
        if (parts.length > 1) {
            finalDisplayValue = parts[1].trim(); // Take the part after "->"
        } else {
            finalDisplayValue = workString; // No "->", so use the (cleaned) string as is
        }

        valueMainDisplay.innerText = finalDisplayValue;

        // Ensure description span is always empty
        if (valueDescDisplay) {
            valueDescDisplay.innerText = "";
        }
    }

    function toggleSection(element) {
        const content = element.nextElementSibling;
        const arrow = element.querySelector('span:last-child');
